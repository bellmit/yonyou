CREATE
    TABLE TM_VEHICLE_ORDER_AUDIT
    (
        DEALER_CODE VARCHAR(30) NOT NULL,
        PO_NO VARCHAR(12) NOT NULL,
        AUDITING_REMARK VARCHAR(300),
        AUDITING_DATE DATETIME,
        AUDITING_BY VARCHAR(50),
        REMARK VARCHAR(400),
        D_KEY DECIMAL(1,0) DEFAULT 0,
        CREATED_BY DECIMAL(14,0),
        CREATED_AT DATETIME,
        UPDATED_BY DECIMAL(14,0),
        UPDATED_AT DATETIME,
        VER DECIMAL(5,0) DEFAULT 0,
        ITEM_ID DECIMAL(14,0),
        AUDITED_BY_IDENTITY DECIMAL(8,0),
        AUDITING_RESULT DECIMAL(8,0),
        SUBMIT_AUDITED_BY DECIMAL(14,0),
        COMMIT_DATE DATETIME,
        DDCN_UPDATE_DATE DATETIME,
         PRIMARY KEY (DEALER_CODE, PO_NO)
    )
