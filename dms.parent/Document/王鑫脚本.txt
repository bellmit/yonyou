DROP TABLE IF EXISTS `TM_TRACKING_TASK`;
CREATE
    TABLE TM_TRACKING_TASK
    (
        DEALER_CODE VARCHAR(30) NOT NULL,
        TASK_ID DECIMAL(14,0) NOT NULL AUTO_INCREMENT,
        CUSTOMER_STATUS DECIMAL(8,0) DEFAULT 13211001,
        INTENT_LEVEL DECIMAL(8,0),
        TASK_NAME VARCHAR(150),
        EXECUTE_TYPE DECIMAL(8,0),
        TASK_CONTENT VARCHAR(150),
        INTERVAL_DAYS DECIMAL(8,0),
        KEEP_DAYS DECIMAL(8,0),
        TASK_TYPE DECIMAL(8,0),
        CONTACTOR_TYPE DECIMAL(8,0),
        IS_VALID DECIMAL(8,0) DEFAULT 12781001,
        HAS_EXECUTE DECIMAL(8,0) DEFAULT 12781002,
        CR_EXECUTE_DATE TIMESTAMP,
        CREATED_BY VARCHAR(30),
        CREATED_AT DATETIME,
        UPDATED_BY VARCHAR(30),
        UPDATED_AT DATETIME,
        VER DECIMAL(5,0) DEFAULT 0,
        BIG_CUSTOMER_INTERVAL_DAYS DECIMAL(8,0) DEFAULT 0,
		PRIMARY KEY (DEALER_CODE, TASK_ID)
    );
